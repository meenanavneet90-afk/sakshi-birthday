<script>
let currentSlide=1;
function nextSlide(){
 document.getElementById("s"+currentSlide).classList.remove("active");
 currentSlide++;
 if(currentSlide>4) return;
 document.getElementById("s"+currentSlide).classList.add("active");
 if(currentSlide==2){ document.getElementById("bgmusic").play(); hearts(); confettiBlast(); }
 if(currentSlide==3){ hearts(); confettiBlast(); }
 if(currentSlide==4){ showTypedMessage(); hearts(); confettiBlast(); }
}

// Auto slide after delay
setTimeout(nextSlide,2000);
setTimeout(nextSlide,6000);
setTimeout(nextSlide,10000);
setTimeout(nextSlide,14000);

function hearts(){
 for(let i=0;i<50;i++){
   let h=document.createElement("div");
   h.className="effect";
   h.innerHTML="❤️";
   h.style.left=Math.random()*100+"%";
   h.style.animationDuration=(2+Math.random()*3)+"s";
   document.body.appendChild(h);
 }
}

function confettiBlast(){
 for(let i=0;i<50;i++){
   let c=document.createElement("div");
   c.className="effect";
   c.innerHTML="🎉";
   c.style.left=Math.random()*100+"%";
   c.style.animationDuration=(2+Math.random()*3)+"s";
   document.body.appendChild(c);
 }
}

function showTypedMessage(){
 const msg = "Jab se tum meri life me ayi ho...\n" +
 "Tum meri duniya ka sabse khoobsurat hissa ho 💖\n" +
 "Main tumhare bina adhura hoon 😘\n\n" +
 "Tumhare saare sapne poore ho,\n" +
 "Aur hamesha khush raho 🌸\n\n" +
 "Happy Birthday Sakshi 🎂💫\n" +
 "Love You 😍❤️";
 let idx = 0;
 let target = document.getElementById("typed");
 function typeIt(){
   if(idx < msg.length){
     target.textContent += msg.charAt(idx);
     idx++;
     setTimeout(typeIt, 50);
   }
 }
 typeIt();
}
</script>
